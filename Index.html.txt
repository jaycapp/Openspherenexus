<script>
// --- CONFIG ---
// Replace with your JSONBin details for persistence
const JSONBIN_URL = "https://api.jsonbin.io/v3/b/YOUR_BIN_ID"; 
const JSONBIN_KEY = "YOUR_SECRET_KEY";

let posts = [];

// Fetch posts from backend
async function fetchPosts() {
  try {
    const res = await fetch(JSONBIN_URL, {
      headers: { "X-Master-Key": JSONBIN_KEY }
    });
    const data = await res.json();
    posts = data.record.posts || [];
    renderFeed();
  } catch (err) { console.error(err); }
}

// Save posts to backend
async function savePosts() {
  try {
    await fetch(JSONBIN_URL, {
      method: "PUT",
      headers: {
        "Content-Type": "application/json",
        "X-Master-Key": JSONBIN_KEY
      },
      body: JSON.stringify({ posts })
    });
  } catch (err) { console.error(err); }
}

// Add a new post
function addPost() {
  const text = document.getElementById('postText').value.trim();
  const imageUrl = document.getElementById('postImage').value.trim();
  const tag = document.getElementById('postTag').value || 'General';
  if (!text && !imageUrl) return alert('Please add text or image!');
  const post = { id: Date.now(), text, image: imageUrl || null, likes:0, comments:[], tag };
  posts.unshift(post);
  document.getElementById('postText').value = '';
  document.getElementById('postImage').value = '';
  renderFeed();
  savePosts();
}

// Like a post
function likePost(id) {
  const post = posts.find(p => p.id === id);
  post.likes++;
  renderFeed();
  savePosts();
}

// Add a comment
function addComment(postId, inputId) {
  const input = document.getElementById(inputId);
  const text = input.value.trim();
  if (!text) return;
  const post = posts.find(p => p.id === postId);
  post.comments.push({ username: 'You', text });
  input.value = '';
  renderFeed();
  savePosts();
}

// Render feed
function renderFeed() {
  const feed = document.getElementById('feed');
  feed.innerHTML = '';
  const filter = document.getElementById('filterTag').value;
  let filteredPosts = posts;
  if (filter !== 'All') filteredPosts = posts.filter(p => p.tag === filter);
  filteredPosts.forEach(post => {
    const div = document.createElement('div');
    div.className = 'post';
    div.innerHTML = `
      <div>${post.text}</div>
      ${post.image ? `<img src="${post.image}" />` : ''}
      <div class="likes-comments">
        <span onclick="likePost(${post.id})">❤️ ${post.likes}</span>
      </div>
      <div class="comment-section">
        ${post.comments.map(c => `<div class="comment">${c.username}: ${c.text}</div>`).join('')}
        <input type="text" id="input-${post.id}" placeholder="Add a comment">
        <button onclick="addComment(${post.id}, 'input-${post.id}')">Post</button>
      </div>
    `;
    feed.appendChild(div);
  });
}

// Initial fetch and auto-refresh every 5s
fetchPosts();
setInterval(fetchPosts, 5000);
</script>
